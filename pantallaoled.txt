void main(void)
{
    OSCCONbits.IRCF = 0b111;
    OSCCONbits.SCS = 0b10;

    ADCON1 = 0x0F;

    I2C_Init();
    __delay_ms(100);

    OLED_Init();
    OLED_Clear();

    OLED_SetCursor(0, 0);
    OLED_String("CONTROL AMBIENTAL");

    OLED_SetCursor(2, 0);
    OLED_String("HUM: 47%");

    OLED_SetCursor(4, 0);
    OLED_String("FAN: ON");

    OLED_SetCursor(6, 0);
    OLED_String("PIC18F4550");

    while (1)
    {
    }
}

void I2C_Init(void)
{
    TRISBbits.TRISB0 = 1;
    TRISBbits.TRISB1 = 1;

    SSPSTAT = 0x80;
    SSPCON1 = 0x28;
    SSPCON2 = 0x00;

    SSPADD = 19;

    PIR1bits.SSPIF = 0;
}

void I2C_Wait(void)
{
    while ((SSPCON2 & 0x1F) || SSPSTATbits.R_nW);
}

void I2C_Start(void)
{
    I2C_Wait();
    SSPCON2bits.SEN = 1;
    while (SSPCON2bits.SEN);
}

void I2C_Stop(void)
{
    I2C_Wait();
    SSPCON2bits.PEN = 1;
    while (SSPCON2bits.PEN);
}

uint8_t I2C_Write(uint8_t data)
{
    I2C_Wait();

    SSPBUF = data;

    while (!PIR1bits.SSPIF);
    PIR1bits.SSPIF = 0;

    return SSPCON2bits.ACKSTAT;
}

void OLED_Command(uint8_t cmd)
{
    I2C_Start();
    I2C_Write(OLED_ADDR_W);
    I2C_Write(0x00);
    I2C_Write(cmd);
    I2C_Stop();
}

void OLED_Data(uint8_t data)
{
    I2C_Start();
    I2C_Write(OLED_ADDR_W);
    I2C_Write(0x40);
    I2C_Write(data);
    I2C_Stop();
}

void OLED_Clear(void)
{
    uint8_t page;
    uint8_t column;

    for (page = 0; page < 8; page++)
    {
        OLED_SetCursor(page, 0);

        for (column = 0; column < 128; column++)
        {
            OLED_Data(0x00);
        }
    }
}

void OLED_SetCursor(uint8_t page, uint8_t column)
{
    OLED_Command(0xB0 + page);
    OLED_Command(0x00 + (column & 0x0F));
    OLED_Command(0x10 + ((column >> 4) & 0x0F));
}

void OLED_Char(char c)
{
    uint8_t i;
    uint8_t index;

    if (c >= 'a' && c <= 'z')
    {
        c = c - 32;
    }

    if (c < 32 || c > 90)
    {
        c = 32;
    }

    index = c - 32;

    for (i = 0; i < 5; i++)
    {
        OLED_Data(font_5x7[index][i]);
    }

    OLED_Data(0x00);
}

void OLED_String(const char *text)
{
    while (*text)
    {
        OLED_Char(*text);
        text++;
    }
}
